1.需求分析

 该设计采用菜单作为应用程序的主要界面，用控制语句来改变程序执行的顺序，控制语句是实现结构化程序设计的基础，通过利用单链表来实现该任务。该设计的任务是通过利用单链表的各项功能来实现一个简单实用的菜单，通过菜单单项进行选择，实现和完成通讯录管理中常用的几个不同的功能。

（4）       菜单内容

1、  通讯录链表的建立

2、  通讯者结点的插入

3、  通讯者结点的查询

4、  通讯者结点的删除

5、  通讯录链表的输出

0、  退出管理系统

请选择0~5：

（5）       设计要求

使用0~5来选择菜单项，其他输入则不起作用。

（6）       功能函数设计

5个不同功能的算法实现编程题，目的是练习利用链表结构来解决实际应用问题的能力，进一步理解和熟悉线形表的链式存储结构。

#include <stdio.h>

typedef struct student

{                                 //通讯录结点类型

} DataType;

  typedef struct node {             //结点类型定义

  DataType data;                  //结点数据域

  struct node *next;              //结点指针域

} ListNode,* LinkList;

void  CreateList(LinkList &head)

{ 

   head=(LinkList)malloc(sizeof(ListNode));

   head->next=NULL;

}

 

//*************通讯录结点的插入***********//

void InsertNode(LinkList &head)

{   

 system("cls");

 printf("************通讯录结点的插入*********\n");

    char  flag ;

 do{ LinkList p,q=head->next,r=head;

 printf("请插入信息\n");

 p=(LinkList)malloc(sizeof(ListNode ));   //申新结点

 scanf("%s",p->data.phone);

//*************通讯者结点的删除*************//

 

void DelNode(LinkList &head)

{

 system("cls");

 printf("********* 通讯录信息的删除 **********\n");

 char jx;

 char flag;

    LinkList p,q=head;

 if(!head->next)

  printf("该链表为空请向链表中插入记录!\n");

 else

 {

  do{

   p=ListFind(head); //调用查找函数

   if (p==NULL)

   {

    printf(" 没有查到要删除的通讯者!\n"); 

    goto jump;

   }

   printf(" 真的要删除该结点吗?(y/n)\n:");

   scanf("%*c%c",&jx);

   if (jx=='y'||jx=='Y')

   {

    while (q &&q->next!=p)

    q=q->next;

    q->next=p->next; //删除结点

free(p); //释放被删结点空间

printf(" 通讯者已被删除\n");

   }

jump: printf("是否继续删除? (y/n)\n");

   scanf("%*c%c",&flag);

 }while(flag=='Y'||flag=='y');

 }

 printf("输入 0:结束;  输入5:返回主菜单 : "); 

}

 

//***************通讯录链表的查询**************// 

void print(DataType e)

{

 printf("   编号     姓名      性别        电话         QQ \n");

 printf("%-12s%-10s%-11s%-14s%-8s\n",e.num,e.name,e.sex,e.phone,e.QQ);

}

void SearchNode(LinkList head,void (*vi)(DataType ))

{

 system("cls");

 printf("********** 通讯录信息的查询 ***********\n");

 char flag;

 LinkList p;

 if(!head->next)

  printf("该链表为空请向链表中插入记录!\n");

 else

 {

  do{

  p=ListFind(head);

  if(p==NULL)

  {

   printf("   没有查到要查询的通讯者! \n");

  }

 

 else

  {

   vi(p->data);

  }

  printf("是否继续查询?(y/n)");

  scanf("%*c%c",&flag);

  }while(flag=='y'||flag=='Y');

 }

 printf("输入 0:结束;  输入5:返回主菜单 : ");

}

 

//**************通讯录链表的输出**************// 

void PrintList(LinkList head)

{

 system("cls");

 LinkList p;

 p=head->next;

 if(!head->next)

  printf("该链表为空请向链表中插入记录!\n");

 else

 {

   printf("   编号     姓名      性别        电话         QQ \n");

 printf("                                                      \n");

 while (p!=NULL)

 {

 printf("%-12s%-10s%-11s%-14s%-8s\n",p->data.num,p->data.name,p->data.sex,p->data.phone,p->data.QQ);

 printf("                                                                            \n");

 p=p->next;                                          //后移一个结点

 }

 }

 printf("输入 0:结束;  输入5:返回主菜单 : ");

}

 

void main()

{   int i;

 LinkList head;

 CreateList(head);

    menu_select();

 scanf("%d",&i);

 while(i)

 {

 {

  case 2:SearchNode(head,print);break;  //通讯录信息的查询

  case 3:DelNode(head);break;//通讯录信息的删除 

  case 4:PrintList(head);break;//通讯录链表的输出 

  case 5:menu_select();break;

 }

  scanf("%d",&i);

 }

}

5.课程设计总结

经过两周的编程，我成功的用链表的方法编写出一套完整的通讯录管理系统。在通过这次的课程设计，使我更进一步的了解认识数据结构这门学科。通过之前C语言打下的基础，在这次的课程设计中让我轻松了不少。相对于C语言来说，数据结构需要更深度的理论和完整准确的数据作为支撑点，每个程序的编译都要掌握其链表的操作方法,明确它的目的.通过这次数据结构课程设计，使我加深了对链表的了解，也提高了对链表的应用能力，这个题目是通过由好几个模块组成，所以我们必须用自己平时所学的和发挥自己的创造性的思维去完成一个程序。通过这次课程设计使我懂得了理论与实际相结合是很重要的，只有理论知识是远远不够的，只有把所学的理论知识与实践相结合起来，从理论中得出结论，才能真正为社会服务，从而提高自己的实际动手能力和独立思考的能力。在设计的过程中遇到问题，可以说得是困难重重，这毕竟第一次做的，难免会遇到过各种各样的问题，同时在设计的过程中发现了自己的不足之处，对以前所学过的知识理解得不够深刻，掌握得不够牢固，通过这次课程设计之后，一定把以前所学过的知识重新温故。同时，对于能够顺利地这次课程设计，
