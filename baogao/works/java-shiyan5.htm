class bankcount

{

       int id;

       String name,date;

       float money;

       public bankcount(int id,String name,String date,float money) {

              this.id=id;

              this.name=name;

              this.date=date;

              this.money=money;

       }

}

class bcoption

{

       Vector vec=new Vector();

       int count=0;

       public void kaihu(bankcount bc) {

              count++;

              vec.add(bc);

       }

      

       public void moneyout(int id,float outmoney) {

              bankcount bc=(bankcount)vec.get(id);

              bc.money=bc.money-outmoney;

              vec.set(id, bc);

       }

      

       public void moneyin(int id,float inmoney) {

              bankcount bc=(bankcount)vec.get(id);

              bc.money+=inmoney;

              vec.set(id,bc);

       }

       public void transfer(int idout,int idin,float money){

              moneyin(idin, money);

              moneyout(idout, money);

       }

       public void query(int id) {

              bankcount bc=(bankcount)vec.get(id);

              System.out.println("["+bc.id+"]\t name:"+bc.name+"\t money:"+bc.money+"￥");

       }

}
