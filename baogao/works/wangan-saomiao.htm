3．实验原理（含设计）(功能组成示意图或框图)

总体设计：

参照《计算机网络安全教程》第四章“网络扫描与监听”

原理：参照书P106页（班长分享提供）：

 

①　网络扫描的原理是利用操作系统提供的connect（）系统调用，与每一个感兴趣的目标计算机的端口进行连接。如果端口处于工作状态，那么connect（）就能成功。这个技术的一个最大优点是，不需要任何权限，系统中的任何用户都有权利是用这个调用。

 

②　扫描的目的是利用各种工具对攻击目标的IP地址或地址段的主机查找漏洞。扫描采取模拟攻击的形式对目标可能存在的已知安全漏洞逐项进行检查。目标可以使工作站，服务器，交换机，路由器和数据库应用等。根据扫描结果向扫描者或管理员提供周密可靠的分析报告。

 

③　主动式扫描一般可以分成：活动主机探测，ICMP查询，网络ping扫描，端口扫描，识别UDP和TCP服务，指定漏洞扫描，总和扫描。

④　扫描方式可以分为两大类：慢速扫描和乱序扫描。

 

设计：参照书P110页，利用TCP协议实现端口扫描

 

①　实现端口扫描的程序可以使用TCP协议和UDP协议

 

②　总体设计是利用TCP连接进行测试。

 

③　通过socket连接计算机的某个端口，试图和该端口建立连接，如果建立成功，就说明对方开放了该端口，如果失败了，就说明对方没有开放端口。

详细设计：

  端口扫描代码（proj4_4.cpp）：

#include<stdio.h>

#include<stdlib.h>

#include<winsock2.h>

#include <conio.h>

#include <windows.h>

//动态链接库

#pragma comment(lib,"ws2_32.lib")

//targetIPAddr目标主机IP

char TargetIPAddr[16];

unsigned int StartPort;  //起始端口

unsigned int EndPort;   //目标端口

 

//声明扫描信息函数

void InputScanInfo(void);

int main(int argc,char **argv){

    InputScanInfo();

    //定义socket

    SOCKET ScanSocket;

    //定义IPV4地址参数

    struct  sockaddr_in TargetAddr_in;

   int Ret;

   //获取标准输出设备句柄

   HANDLE hCon = GetStdHandle(STD_OUTPUT_HANDLE);

   //获得控制台窗口的文本属性

   CONSOLE_SCREEN_BUFFER_INFO bInfo;

   GetConsoleScreenBufferInfo(hCon,&bInfo);

   //定义wsaData用于存放windows cocket 初始化信息

   WSADATA wsaData;

   //初始化Winsock

   if((Ret = WSAStartup(MAKEWORD(2,1),&wsaData))!=0){

       printf("WSAStartup failed with error %d\n",Ret);

       return 0;

   }

   //GetTickCount()获取系统启动到现在的毫秒时间

   DWORD dwStart = GetTickCount();

   printf("Target IP:%s\n",TargetIPAddr);

   //端口循环扫描过程

   for(unsigned int i = StartPort;i<=EndPort;i++){

       //创建socket,IPV4地址族，TCP

       ScanSocket = socket(AF_INET,SOCK_STREAM,IPPROTO_TCP);

       if ((ScanSocket == INVALID_SOCKET)) {

           printf("sock failed with error:%d\n",WSAGetLastError());

           return 0;

       }

       //设置IP地址参数

       TargetAddr_in.sin_family = AF_INET;

       TargetAddr_in.sin_addr.s_addr = inet_addr(TargetIPAddr);

       TargetAddr_in.sin_port = htons(i);

       //使用connect（）连接服务器

       if(connect(ScanSocket,(struct sockaddr*)&TargetAddr_in,sizeof(TargetAddr_in))==SOCKET_ERROR){

           //连接失败，绿色提示信息

           SetConsoleTextAttribute(hCon,10);

           printf("Port%5d Close\n",i);

       }

       else

       {

           //连接成功，黄色提示信息

           SetConsoleTextAttribute(hCon,14);

           printf("Port%5d Open\n",i);

       }

       //关闭socket,释放志愿

       if(closesocket(ScanSocket)==SOCKET_ERROR){

           printf("closesocket failed with error %d\n",WSAGetLastError());

           return 0;

       }

   }

   //打印扫描总耗时

    SetConsoleTextAttribute(hCon,14);

    printf("\ntime:%dms\n",GetTickCount()-dwStart);

    SetConsoleTextAttribute(hCon,bInfo.wAttributes);

    //停winsock dll调用

    if(WSACleanup() == SOCKET_ERROR){

        printf("WSACleanup faailed with error %d\n",WSAGetLastError());

        return 0;

    }

    return 1;

}

 

void InputScanInfo(void){

    //判断参数是否首次输入

    bool abc = true;

    printf("请输入扫描信息------\n");

    //TargetAddr

    do{

        if(abc){

            abc = false;

            printf("请输入目标主机IP地址： ");

            scanf("%s",TargetIPAddr);

        }

        else

        {

            printf("请重新输入: ");

            scanf("%s",TargetIPAddr);

        }

        }while(inet_addr((const char*)TargetIPAddr)==INADDR_NONE);

        //获取StartPort

        abc = true;

        do

        {

           if(abc){

            abc = false;

            printf("请输入UDP扫描起始端口（0-65535）： ");

            scanf("%d",&StartPort);

        }

        else

        {

            printf("请重新输入: ");

            scanf("%d",&StartPort);

        }

        } while (StartPort<0||StartPort>65535);

        //获取EndPort

            abc = true;

        do

        {

           if(abc){

            abc = false;

            printf("请输入UDP扫描结束端口（0-65535）： ");

            scanf("%d",&EndPort);

        }

        else

        {

            printf("请重新输入: ");

            scanf("%d",&EndPort);

        }

        } while (StartPort<StartPort||StartPort>65535);

        //输入完成后清屏

        system("cls");

        printf("UDP扫描信息-------:\n（目标主机IP地址：端口范围）%s:%d-%d\n",TargetIPAddr,StartPort,EndPort);

    

}
